47  /***
48   * This class is the base class for all expressions that are binary
49   * operators.
50   *
51   * @version     $Revision: 1.4 $ $Date: 2003/08/09 15:49:50 $
52   * @author      <a href="mailto:tma@netspace.net.au">Tim Anderson</a>
53   * @see         Expression
54   * @see         Operator
55   */
56  abstract class BinaryOperator extends Operator {
57  
58      /***
59       * The left-hand side of the binary expression
60       */
61      private final Expression _lhs;
62  
63      /***
64       * The right-hand side of the binary expression
65       */
66      private final Expression _rhs;
67  
68      /***
69       * Construct a new <code>BinaryOperator</code>
70       *
71       * @param operator the operator
72       * @param lhs the left hand side of the expression
73       * @param rhs the right hand side of the expression
74       */
75      public BinaryOperator(final String operator, final Expression lhs,
76                            final Expression rhs) {
77          super(operator);
78          _lhs = lhs;
79          _rhs = rhs;
80      }
81  
82      /***
83       * Returns the left hand side of the expression
84       *
85       * @return the left hand side of the expression
86       */
87      public final Expression left() {
88          return _lhs;
89      }
90  
91      /***
92       * Returns the right hand side of the expression
93       *
94       * @return the right hand side of the expression
95       */
96      public final Expression right() {
97          return _rhs;
98      }
99  
100     /***
101      * Returns a stringified version of the expression
102      *
103      * @return a stringified version of the expression
104      */
105     public final String toString() {
106         StringBuffer result = new StringBuffer();
107         result.append('(');
108         result.append(_lhs.toString());
109         result.append(' ');
110         result.append(operator());
111         result.append(' ');
112         result.append(_rhs.toString());
113         result.append(')');
114         return result.toString();
115     }
116 
117 } //-- BinaryOperator
