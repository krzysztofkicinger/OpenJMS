52  import java.io.ByteArrayInputStream;
53  import java.io.ByteArrayOutputStream;
54  import java.io.IOException;
55  import java.io.ObjectInputStream;
56  import java.io.ObjectOutputStream;
57  
58  
59  /***
60   * A utility class to assist with serialization and desearialization of objects
61   * to and from byte stream
62   */
63  public class SerializationHelper {
64  
65      /***
66       * Serialize the specified object
67       *
68       * @param object - object to serialize
69       * @return byte[] - the serialized object
70       * @throws IOException - if there is an error with serialization
71       */
72      static public byte[] serialize(Object object)
73          throws IOException {
74  
75          ByteArrayOutputStream bstream = new ByteArrayOutputStream();
76          ObjectOutputStream ostream = new ObjectOutputStream(bstream);
77          ostream.writeObject(object);
78  
79          return bstream.toByteArray();
80      }
81  
82      /***
83       * Deserialise the specified blob into an Object and return it
84       * to the client.
85       * <p>
86       * If there is an error with the deserialization then throw an
87       * IOException
88       *
89       * @param blob - the blob of bytes to deserialize
90       * @return Object - the object to return
91       * @throws IOException - if it cannot deseriliaze
92       * @throws IllegalArgumentException if the blob is null
93       */
94      static public Object deserialize(byte[] blob)
95          throws IOException, ClassNotFoundException {
96  
97          if (blob == null) {
98              throw new IllegalArgumentException("null blob to deserialize");
99          }
100 
101         ByteArrayInputStream bstream = new ByteArrayInputStream(blob);
102         ObjectInputStream istream = new ObjectInputStream(bstream);
103 
104         return istream.readObject();
105     }
106 }
