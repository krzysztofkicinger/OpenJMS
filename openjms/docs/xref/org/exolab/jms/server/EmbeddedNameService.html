46  package org.exolab.jms.server;
47  
48  import javax.naming.Context;
49  
50  import org.apache.avalon.excalibur.naming.DefaultNameParser;
51  import org.apache.avalon.excalibur.naming.DefaultNamespace;
52  import org.apache.avalon.excalibur.naming.memory.MemoryContext;
53  
54  import org.exolab.jms.client.JmsConnectionFactoryBuilder;
55  import org.exolab.jms.client.JmsDestinationFactory;
56  
57  
58  public class EmbeddedNameService {
59  
60      /***
61       * The initial context
62       */
63      private Context _context;
64  
65      /***
66       * The singleton instance
67       */
68      private static EmbeddedNameService _instance = new EmbeddedNameService();
69  
70  
71      /***
72       * Construct an <code>EmbeddedNameService</code>
73       */
74      private EmbeddedNameService() {
75          DefaultNameParser parser = new DefaultNameParser();
76          DefaultNamespace namespace = new DefaultNamespace(parser);
77          namespace.addObjectFactory(new JmsConnectionFactoryBuilder());
78          namespace.addObjectFactory(new JmsDestinationFactory());
79          _context = new MemoryContext(namespace, null, null);
80      }
81  
82      /***
83       * Returns the singleton instance of the embedded name service
84       *
85       * @return the singleton instance of the embedded name service
86       */
87      public static EmbeddedNameService getInstance() {
88          return _instance;
89      }
90  
91      /***
92       * Returns the initial context for beginning name resolution
93       *
94       * @return the initial context
95       */
96      public Context getInitialContext() {
97          return _context;
98      }
99  
100 } //-- EmbeddedNameService
